配置文件说明
```
server = "10.100.61.3" #认证服务器，可以使用域名auth.jlu.edu.cn
username = "" #用户名，和客户端一样
password = "" #密码，和客户端一样
host_name = "liyuanyuan" #计算机名，不要超过71个字符
host_os = "Windows 4.0" #操作系统，不要超过128个字符
mac = 0x888888888888 #网络中心上注册时IP对应的MAC
```
